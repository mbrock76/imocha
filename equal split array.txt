import java.util.*;
import java.io.*;
import java.lang.Math;

public class Main
{
  public static int equalSplitArray(int N,int[] A)
  {

    //this is default OUTPUT. You can change it.
    int index = -1;
    String leftNums = "";
    String rightNums = "";
    int leftCount = 0;
    int rightCount = 0;

    //write your Logic here:
    for(int i = 1; i < N; i++){

      leftNums = "";
      rightNums = "";
      leftCount = 0;
      rightCount = 0;

      for(int left = 0; left < i; left++){

        if(!leftNums.contains(Integer.toString(A[left]))){
          leftCount++;
        }
        leftNums += Integer.toString(A[left]);
      }

      for(int right = i; right < N; right++){

        if(!rightNums.contains(Integer.toString(A[right]))){
          rightCount++;
        }
        rightNums += Integer.toString(A[right]);
      }

      if(leftCount == rightCount){

        index = i;
        break;
      }
    }

      return index;
  }

  public static void main (String[]args)
  {
    Scanner sc = new Scanner (System.in);

    //INPUT [uncomment & modify if required]
    int N = sc.nextInt ();

    int[] A = new int[N];
    for (int i = 0; i < N; i++)
    {
        A[i] = sc.nextInt ();
    }
    
    sc.close ();

    //OUTPUT [uncomment & modify if required]
    System.out.print(equalSplitArray(N,A));
  }
}
